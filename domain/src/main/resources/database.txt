create table Address (
id bigint not null auto_increment, 
city varchar(255), 
country varchar(255), 
landMark varchar(255), state varchar(255), 
zip varchar(255),
primary key (id));


create table Location (
id bigint not null auto_increment, 
lat double precision not null, 
lng double precision not null, 
addressId bigint , 
venueId bigint,
primary key (id),
foreign key (addressId) references Address (id)
foreign key (venueId) references Venue (id));


create table Category (
id bigint not null auto_increment, 
name varchar(255), 
primary key (id));


create table Venue (
id bigint not null auto_increment, 
image varchar(255), 
name varchar(255) not null, 
verified varchar(255), 
addressId bigint, 
categoryId bigint, 
primary key (id),
foreign key (addressID) references Address (id),
foreign key (categoryId) references Category (id));


create table Groups (
id bigint not null auto_increment,
name varchar(255), 
primary key (id));


create table Merchant (
id bigint not null auto_increment, 
loginId varchar(255), 
name varchar(255), 
pwd varchar(255), 
groupId bigint, 
venueId bigint, 
primary key (id),
foreign key (groupId) references Groups (id),
foreign key (venueId) references Venue (id));


create table Offer (
id bigint not null auto_increment, 
startingDate date, 
EndingDate date, 
description varchar(255), 
venueId bigint, 
primary key (id),
foreign key (venueId) references Venue (id));



create table Shout (
id bigint not null auto_increment, 
desciption varchar(255), 
offerId bigint, 
venueId bigint, 
primary key (id),
foreign key (offerId) references Offer (id),
foreign key (venueId) references Venue (id));
